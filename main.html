// script.js // This script dynamically adds the dog images and links to the list document.addEventListener("DOMContentLoaded", function() { const dogList = document.createElement("ul"); // Dog data const dogs = [ { name: "Golden Retriever", img: "https://upload.wikimedia.org/wikipedia/commons/5/5f/Golden_Retriever_medium-to-light-coat.jpg", link: "https://en.wikipedia.org/wiki/Golden_Retriever" }, { name: "Poodle", img: "https://upload.wikimedia.org/wikipedia/commons/7/70/Toy_Poodle_black.jpg", link: "https://en.wikipedia.org/wiki/Poodle" } ]; // Create list items dynamically dogs.forEach(dog => { const li = document.createElement("li"); const a = document.createElement("a"); const img = document.createElement("img"); a.href = dog.link; a.target = "_blank"; img.src = dog.img; img.alt = dog.name; img.width = 300; a.appendChild(img); li.appendChild(a); dogList.appendChild(li); }); document.body.appendChild(dogList); });
